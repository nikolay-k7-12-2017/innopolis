CREATE TABLE users
(
  id       INTEGER NOT NULL
    CONSTRAINT users_id_pk
    PRIMARY KEY,
  login    VARCHAR(200),
  password VARCHAR(200),
  enabled  BOOLEAN,
  role     VARCHAR(200)
);

CREATE TABLE mimimi
(
  id      INTEGER NOT NULL
    CONSTRAINT mimimi_pkey
    PRIMARY KEY,
  picture VARCHAR NOT NULL,
  name    VARCHAR NOT NULL,
  counter INTEGER NOT NULL
);
